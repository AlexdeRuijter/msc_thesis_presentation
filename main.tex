\begin{frame}{What are we learning}
    \begin{theorem}
    [Universal Approximation of Fading Memory Operators by Volterra Series, \texorpdfstring{\cite[Theorem 4]{BoydChua1985}}{Boyd and Chua \textit{(1975)} Theorem 4}]
    \label{thm:volterra_universal_approximation}
    Let $\epsilon >0$ and 
    \begin{equation}
        \Kcal \coloneqq \{ \vc{u} \in \ell^\infty : \norm{\vc{u}}_{\infty} \leq M \}.
    \end{equation}
    Suppose that $\mx{N}$ is \emph{any} causal time-invariant operator $: \ell^\infty \to \ell^\infty$ with fading memory on $\Kcal$.
    Then, there exists a polynomial $p: \Rbb^m \to \Rbb$ such that for all $\vc{u} \in \Kcal$,
    \begin{equation}
        \norm{\mx{N}\vc{u} - \mx{P}\vc{u}}_\infty < \epsilon,
    \end{equation}
    and $\mx{P}u(k) = p(u(k), u(k-1), \dots, u(k-m+1))$.
\end{theorem}
\end{frame}

\begin{frame}{Volterra Series: A Universal Representation}
    \begin{definition}[Discrete-Time Volterra Model]
    \label{def:volterra}
    Let $u_t \in \Rbb$ be the input and $y_t \in \Rbb$ be the output at time $t$.
    A Volterra model of degree $P$ with memory $M$ is defined as the expansion
    \begin{equation}
        y_t = h_0 + \sum_{p=1}^P \sum_{\tau_1=0}^{M-1} \dots \sum_{\tau_p=\tau_{p-1}}^{M-1} h_p(\tau_1, \dots, \tau_p) \prod_{j=1}^p u_{t-\tau_j} + \epsilon_t
    \end{equation}
    where $h_p$ are the Volterra kernels and $\epsilon_t$ is an additive noise process.
\end{definition}
\end{frame}

\begin{frame}{Shining a Light on Excitation}
    Goal of the slide:
    Compare excitation criteria by using a room and light analogy
    \begin{itemize}
        \item Classical PE: Asymptotic, every direction eventually excited, long shutter times needed to see anything.
        \item BMSB: Blockwise, we need to see the room once in every block, we can not stay fixed on a corner for a long time.
        \item dPE: directional (flashlight), probabilistic (we don't need fixed movements) - with high probability we can understand what is in the room over a finite time window, even if we only see one corner at a time, as long as we see all corners often enough.
    \end{itemize}
\end{frame}

\begin{frame}{Main Result: Non-Asymptotic Bounds for Volterra Learning}
\begin{theorem}[Sample Complexity of Volterra Models]
\label{thm:volterra_sample_complexity}
Consider a Volterra model of memory $M$ and degree $P$ with dimension $D = \binom{M+P}{P}$. 
There exists an absolute constant $C>0$ such that if
\begin{equation}
N \geq \frac{C}{\beta}\max\left\{\; D\log\frac{2}{\beta},\;\log\frac{2}{\delta}\;\right\},
\end{equation}
then with probability at least $1 -\delta$,
\begin{equation}
\|\hat{\theta}_N - \theta^\star\|_2 \leq \frac{K}{\sqrt{\alpha \beta N}} \left(\sqrt{D} + \sqrt{\log \frac{2}{\delta}}\right).
\end{equation}
\end{theorem}
\end{frame}
